// Анализ массива данных
const data = [12, 45, 7, 23, 56, 89, 34];

const stats = {
  mean: data.reduce((a, b) => a + b) / data.length,
  max: Math.max(...data),
  min: Math.min(...data),
};

console.log(`Mean: ${stats.mean.toFixed(2)}`);
console.log(`Max: ${stats.max}, Min: ${stats.min}`);
