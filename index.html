-- Monthly Cohort Retention Analysis
WITH first_purchase AS (
    SELECT
        customer_id,
        DATE_TRUNC('month', MIN(purchase_date)) AS cohort_month
    FROM orders
    GROUP BY customer_id
),
monthly_activity AS (
    SELECT
        o.customer_id,
        fp.cohort_month,
        DATEDIFF('month', fp.cohort_month,
                 DATE_TRUNC('month', o.purchase_date)) AS month_index
    FROM orders o
    JOIN first_purchase fp ON o.customer_id = fp.customer_id
)
SELECT
    cohort_month,
    month_index,
    COUNT(DISTINCT customer_id)                           AS active_users,
    ROUND(
        COUNT(DISTINCT customer_id) * 100.0
        / FIRST_VALUE(COUNT(DISTINCT customer_id))
              OVER (PARTITION BY cohort_month
                    ORDER BY month_index), 2
    )                                                     AS retention_rate
FROM monthly_activity
GROUP BY cohort_month, month_index
ORDER BY cohort_month, month_index;
import pandas as pd
import numpy as np

# ── RFM Customer Segmentation ──────────────────────────
df = pd.read_csv('transactions.csv', parse_dates=['date'])
snapshot = df['date'].max() + pd.Timedelta(days=1)

rfm = df.groupby('customer_id').agg(
    recency  =('date',     lambda x: (snapshot - x.max()).days),
    frequency=('order_id', 'nunique'),
    monetary =('revenue',  'sum')
).reset_index()

# Score each dimension 1–4 (4 = best)
rfm['R'] = pd.qcut(rfm['recency'],  4, labels=[4,3,2,1]).astype(int)
rfm['F'] = pd.qcut(rfm['frequency'].rank(method='first'),
                   4, labels=[1,2,3,4]).astype(int)
rfm['M'] = pd.qcut(rfm['monetary'], 4, labels=[1,2,3,4]).astype(int)
rfm['rfm_score'] = rfm['R']*100 + rfm['F']*10 + rfm['M']

segments = {(4,4,4): 'Champions', (3,4,4): 'Loyal',
            (4,1,1): 'Recent',    (1,1,1): 'Lost'}

print(rfm[['customer_id','recency','frequency','monetary','rfm_score']].head())
print(f"\nChampions: {(rfm['rfm_score'] >= 400).sum()} customers")
